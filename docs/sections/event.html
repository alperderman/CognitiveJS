In CognitiveJS events are defined with cog-event attribute and they will stay after render is completed. Tokens and conditional rendering is allowed inside the attribute to make dynamic changes possible.

    
        <html>
        <head>
            <title>CognitiveJS</title>
        </head>
        <body>

            <button cog-event="click:'clickButton()', mouseover:'hoverButton()'>Send</button>

            <script src="../dist/cog.js"></script>
            <script>
                function clickButton() {
                    console.log("click");
                }
                function hoverButton() {
                    console.log("hover");
                }
                cog.render();
            </script>

        </body>
        </html>
