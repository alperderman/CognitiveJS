Escape definition is possible with cog-escape attribute. With this attribute, elements and their children will be completely excluded from the rendering process.

    
        <html>
        <head>
            <title>CognitiveJS</title>
        </head>
        <body>

            <div cog-escape>
                <p cog-prop="style:{color:{{color}}}">{{hello}}</p>
            </div>

            <script src="../dist/cog.js"></script>
            <script>
                cog.data.hello = "Hello World";
                cog.data.color = "#f00";
                cog.render();
            </script>

        </body>
        </html>
